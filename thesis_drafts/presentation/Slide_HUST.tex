\begin{frame}[c]{Cơ chế 1: Single-Writer + Cơ sở lý thuyết CAP/SMR}
	\begin{alertblock}{Tại sao Single-Writer?}
		\footnotesize
		\begin{enumerate}
		\setlength{\itemsep}{0pt}
			\item Tiến hóa nhóm = \textbf{SMR}: mỗi Commit là một chuyển trạng thái
			\item Commit \alert{không giao hoán}: thứ tự xử lý quyết định kết quả
			\item \textbf{CAP}: P2P phân vùng $\Rightarrow$ chọn \textbf{AP + Eventual Consistency}; CP đóng băng, không phù hợp chat
		\end{enumerate}
	\end{alertblock}
	\vspace{0.05em}
	\begin{columns}[c]
		\begin{column}{0.57\textwidth}
			\begin{block}{Bầu chọn Token Holder xác định}
				\scriptsize
				\begin{gather*}
				\text{Eligible}(E) = \text{Members}(E) \cap \text{ActiveView}(E) \setminus \text{Susp}(E) \\[0.4em]
				\text{TokenHolder}(E) = \argmin_{n \in \text{Eligible}(E)} H(\text{group\_id} \Vert E \Vert n)
				\end{gather*}
			\vspace{0.3em}
			\centering\scriptsize Cùng đầu vào $\to$ cùng kết quả \;|\; $O(N)$ cục bộ
			\end{block}
		\end{column}
		\begin{column}{0.38\textwidth}
		\scriptsize
		\begin{itemize}
		\setlength{\itemsep}{0pt}
			\item \textbf{Proposal:} mọi thành viên tạo
			\item \textbf{Commit:} chỉ Token Holder gom/phát
			\item \textbf{Failover:} ngoại tuyến $\to$ loại ActiveView (Adaptive Lease + $\varphi$-Accrual --- A)
		\end{itemize}
		\end{column}
	\end{columns}
\end{frame}

\begin{frame}[c]{Cơ chế 3: Fork Detection --- Phân biệt chậm vs phân nhánh}
	\begin{columns}[c]
		\begin{column}{0.45\textwidth}
			\small \textbf{History Hash}
			\begin{gather*}
			R(0) = r_0, \quad R(e) = H(R(e{-}1) \Vert C(e)) \\
			\text{\scriptsize $R(E)$ tóm tắt chuỗi Commit } 1{\to}E

			\small
			\begin{itemize}

\begin{frame}[c]{Cơ chế 4: HLC --- Thứ tự hiển thị cho tin nhắn ứng dụng}
	\begin{columns}[c]
		\begin{column}{0.47\textwidth}
			\small \textbf{Vấn đề: Thứ tự hiển thị}
			\begin{itemize}
				\small
				\item \textbf{2 loại tin:} Control (Proposal/Commit) vs Application (chat)
				\item \textbf{Có DS:} DS cấp timestamp
				\item \textbf{P2P:} clock drift $\to$ không tin physical clock
			\end{itemize}
		\end{column}
		\begin{column}{0.49\textwidth}
			\small \textbf{Giải pháp: Hybrid Logical Clock}
			\begin{alertblock}{}
				\small
				$HLC = (L, C, \text{NodeID})$ \\[0.2em]
				\texttt{HLC\_Now():} \\
				\quad $L' = \max(L_{local}, L_{physical})$ \\
				\quad if $L'{=}L_{local}$: $C{+}{+}$ else $C{=}0$ \\
				\quad return $(L', C, \text{NodeID})$
			\end{alertblock}
			\vspace{0.2em}
		\small
		$\bullet$ \textbf{Áp dụng:} application message\\
		$\bullet$ \textbf{Đảm bảo:} happens-before + total order mỗi nút
		\end{column}
	\end{columns}
	\vspace{0.3em}
	\begin{block}{}
		\small \textbf{Tách 2 thứ tự:} mật mã (epoch, Commit) $\neq$ hiển thị (HLC)
	\end{block}
\end{frame}

\begin{frame}[c]{Phụ lục A. Failover an toàn trong Single-Writer}
	\begin{columns}[c]
		\begin{column}{0.48\textwidth}
			\begin{alertblock}{Vấn đề}
				\small
				Timeout cố định 4s dễ khiến nút suspend nhầm Token Holder khi GossipSub jitter, gây fork ngay cả khi mạng vẫn liên thông.
			\end{alertblock}
			\vspace{0.2em}
			\begin{block}{Adaptive Epoch Lease}
				\small
				\begin{gather*}
				L(E) = W_{batch} + \widehat{C}_{commit} + 2\widehat{RTT}_{p95}
				\end{gather*}
				\footnotesize
				Lease tính từ thời gian gom Proposal, tạo Commit và độ trễ p95 của mạng --- thay thế timeout cố định.
			\end{block}
		\end{column}
		\begin{column}{0.48\textwidth}
			\begin{block}{$\varphi$-Accrual Failure Detector}
				\small
				\begin{gather*}
				\varphi(t) = -\log_{10}\bigl(1 - F_{HB}(t - t_{last})\bigr)
				\end{gather*}
				\footnotesize
				$\varphi = 1$: 10\% còn sống (chậm) \newline
				$\varphi = 8$: $10^{-8}$ còn sống (chết)
			\end{block}
			\vspace{0.2em}
			\begin{alertblock}{Điều kiện suspend}
				\small
				\begin{gather*}
				\text{Suspend}(H) \iff L_{node}(E) \text{ hết} \;\wedge\; \varphi > 8
				\end{gather*}
				\footnotesize
				Chỉ suspend khi lease hết \textbf{và} xác suất holder chết cực cao.
			\end{alertblock}
		\end{column}
	\end{columns}
\end{frame}
